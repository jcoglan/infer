extends ./9-1-pure-simply-typed-lambda-calculus.txt

syntax {
  $t := ...
      | true
      | false
      | if $t then $t else $t

  $T := ...
      | Bool
}

rule T-True {
  $Γ ⊢ true : Bool
}

rule T-False {
  $Γ ⊢ false : Bool
}

rule T-If {
  $Γ ⊢ $t1 : Bool; $Γ ⊢ $t2 : $T; $Γ ⊢ $t3 : $T
  ---------------------------------------------
       $Γ ⊢ (if $t1 then $t2 else $t3) : $T
}
